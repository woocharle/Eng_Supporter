

1. 테이블 및 시퀀스 생성 


CREATE TABLE petro(
	idx INT NOT NULL PRIMARY KEY,
	unit VARCHAR(50),
	img VARCHAR(20),
	content LONGTEXT
);



create sequence petro_seq
increment by 1
start with 1;


2. Data입력

insert into petro
values(NEXT VALUE FOR petro_seq, #{unit}, #{content_s}, #{img}, #{content})
